% ---- A.1 Monte Carlo Path Simulation --------------------------------
\subsection*{A.1\quad Monte Carlo Path Simulation}

The function below simulates $M$ asset price paths of length $N+1$ under the
Black--Scholes model, using the exact log-normal discretisation to avoid Euler
approximation bias.

\begin{lstlisting}[caption={Black--Scholes Monte Carlo path simulation.},
                   label={lst:mc}]
import numpy as np

def simulate_bs_paths(S0, r, sigma, T, N, M, seed=42):
    """
    Simulate M risk-neutral GBM paths, each of length N+1.

    Parameters
    ----------
    S0    : float -- initial asset price
    r     : float -- continuously compounded risk-free rate
    sigma : float -- volatility
    T     : float -- maturity in years
    N     : int   -- number of time steps
    M     : int   -- number of paths

    Returns
    -------
    S : ndarray of shape (M, N+1)
    """
    rng = np.random.default_rng(seed)
    dt  = T / N
    Z   = rng.standard_normal((M, N))          # iid N(0,1) increments

    # Exact log-normal discretisation (no Euler bias)
    log_inc = (r - 0.5 * sigma**2) * dt + sigma * np.sqrt(dt) * Z

    log_S = np.concatenate(
        [np.full((M, 1), np.log(S0)),
         np.log(S0) + np.cumsum(log_inc, axis=1)],
        axis=1
    )
    return np.exp(log_S)                        # shape: (M, N+1)
\end{lstlisting}
